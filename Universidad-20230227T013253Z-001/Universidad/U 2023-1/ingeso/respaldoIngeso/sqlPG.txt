create table usuario(
	id SERIAL PRIMARY KEY,
	name varchar(50) not null,
	last_name varchar(50) not null,
	username varchar(50) not null,
	email varchar(50) not null unique,
	password varchar(50) not null,
	role varchar(50) not null,
	status  boolean, --tinyint(1)
	created_at timestamp,--cambiar a datetime en phpmyadmin
	updated_at timestamp--cambiar a datetime en phpmyadmin
);	

create table product(
	id SERIAL PRIMARY KEY,
	nombre varchar(255) not null,
	cantidad integer not null,
	imagen varchar(1000),
	descripcion varchar(100),
	fecha_ingreso timestamp,
	fecha_actualizada timestamp,
	id_usuario serial,
	id_marca serial,
	id_proveedor serial,
	foreign key (id_usuario) references usuario(id),
	foreign key (id_marca) references marca(id),
	foreign key (id_proveedor) references supplier(id)
);
